<!doctype html>
<html>
    <head>
        <meta charset="utf-8">
        <title>Hello World</title>

        h1 {
            color: #FF0000;
            text-align: center;
        }
        </style>
</head>
<body>
    <h1>Hello World!</h1>
</body>
</html>
